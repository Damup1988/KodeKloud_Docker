# show all running containers
docker ps

# show all containers
docker ps -a

# stop a container
docker stop container_name

# delete a container
docker rm container_name

# execute a command on a container
docker exec container_name "your_command"
docker exec container_name cat /etc/hosts

# get info about particluar container
docker inspect container_name
docker inspect container_id

# get container logs
docker logs container_name
docker logs container_id

# build an image
docker build -f path_to_Dockerfile -t "your tags"

# push an image
docker push damir/my-custom-app

# to see all docker layers with sizes
docker history your_image

# run docker commands remotely
docker -H=remote-docker-engine:2375
docker -H=10.123.2.1:2375 run nginx

# list all networks
docker network ls

# get details about specific network
docket inspect _network_id_

# create a new network
docker network create --driver bridge --gateway 182.18.0.1 --subnet 182.18.0.0/16 custom-isolated-network

# to login your docker registry account
docker login
docker login _url_to_private_registry_

# deploy private docker registry
docker run -d -p 5000:5000 --name registry registry:2

# push an image to private registry
docker image tag my-image localhost:5000/my-image
docker push localhost:5000/my-image
